    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const chart = document.getElementById('chart');
            const xAxis = document.getElementById('xAxis');
            const yAxis = document.getElementById('yAxis');
            const gridLines = document.getElementById('gridLines');
            const reportTitle = document.getElementById('reportTitle');
            const avgReservationsEl = document.getElementById('avgReservations');
            const maxReservationsEl = document.getElementById('maxReservations');
            const totalReservationsEl = document.getElementById('totalReservations');
            const printedDateEl = document.getElementById('printedDate');
            const printedByEl = document.getElementById('printedBy');

            // --- Sample Data Generation ---
            const monthName = 'September';
            const printedBy = 'John Doe';
            const numDays = 30;
            const reservations = [];
            let totalReservations = 0;

            for (let i = 1; i <= numDays; i++) {
                const count = Math.floor(Math.random() * 50) + 10; // Random number between 10 and 59
                reservations.push({ day: i, count: count });
                totalReservations += count;
            }

            const maxReservations = Math.max(...reservations.map(r => r.count));
            const avgReservations = Math.round(totalReservations / numDays);
            const maxChartHeight = 350;
            const yAxisMax = 60; // Set a fixed max for the y-axis to match the style

            // --- HTML Injection ---
            reportTitle.textContent = `Monthly Occupancy Report - ${monthName}`;
            
            // Y-axis labels and grid lines
            const yAxisValues = [yAxisMax, Math.ceil(yAxisMax * 0.75), Math.ceil(yAxisMax * 0.5), Math.ceil(yAxisMax * 0.25), 0];
            yAxis.innerHTML = yAxisValues.map(val => {
                if (val === 0) {
                    return `<div class="y-label-zero"><span class="red-flag"></span>${val}</div>`;
                }
                return `<div class="y-label">${val}</div>`;
            }).join('');
            
            gridLines.innerHTML = yAxisValues.slice(0, -1).map(() => `<div class="grid-line"></div>`).join('');

            // Chart bars and X-axis labels
            let chartHtml = '';
            let xAxisHtml = '';
            reservations.forEach(data => {
                const height = (data.count / yAxisMax) * maxChartHeight;
                chartHtml += `
                    <div class="bar" style="height: ${height}px;">
                        <div class="bar-value">${data.count}</div>
                    </div>
                `;
                xAxisHtml += `<div class="x-label">${data.day}</div>`;
            });

            chart.innerHTML = chartHtml;
            xAxis.innerHTML = xAxisHtml;

            // Summary data
            avgReservationsEl.textContent = avgReservations;
            maxReservationsEl.textContent = maxReservations;
            totalReservationsEl.textContent = totalReservations;
            
            // Footer data
            const now = new Date();
            const formattedDate = `${now.getDate()}/${now.getMonth() + 1}/${now.getFullYear()} ${now.getHours()}:${now.getMinutes()}`;
            printedDateEl.textContent = formattedDate;
            printedByEl.textContent = printedBy;
        });
    </script>
